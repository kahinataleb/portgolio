                     int factorial(int n) {
                                if (n == 0 || n == 1) {
                                  return 1;
                                } else {
                               return n * factorial(n - 1);
                               }
                    }
                              
                       int main() {
                                int n;
                                printf("Entrez un nombre entier positif : ");
                               scanf("%d", &n);
                              
                              if (n < 0) {
                                printf("Le factoriel n'est défini que pour les entiers positifs.\n");
                                } else {
                                 int result = factorial(n);
                                  printf("Le factoriel de %d est : %d\n", n, result);
                               }
                              
                               return 0;
                       }
